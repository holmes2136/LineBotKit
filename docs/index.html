Send buttons template message


    LineClientManager lineManager = new LineClientManager(this.token);

    var template = new ButtonsTemplate()
    {
        text = "text",
        imageAspectRatio = ImageAspectRatioType.Square,
        imageBackgroundColor = "#cccccc",
        imageSize = ImageSizeType.Contain,
        thumbnailImageUrl = "https://thumbnailImageUrl.jpg",
        title = "title",
        defaultAction = new UriAction()
        {
            label = "label",
            uri = "http://www.yahoo.com.tw"
        },
        actions = new List(){
            new DatetimePickerAction(){
                    label = "test",
                    data = "abc=aaa&def=111",
                    mode = DateTimePickerMode.Time
            }
        }
    };

    ResponseItem response = await lineManager.PushMessage(
    new PushMessageRequest(){
        to = "User Iden",
        messages = new List(){
                new TemplateMessage("altText",template)
            }
    });

Send confirm template message


    LineClientManager lineManager = new LineClientManager(this.token);

    var template = new ConfirmTemplate()
    {
        text = "text",
        actions = new List(){
            new MessageAction(){
                label = "test",
                text = "text"
            },
            new MessageAction(){
            label = "test",
            text = "text"
            }
        }
    };

    ResponseItem response = await lineManager.PushMessage(
    new PushMessageRequest()
    {
        to = "User Iden",
        messages = new List() {
            new TemplateMessage("altText",template)
        }
    });
